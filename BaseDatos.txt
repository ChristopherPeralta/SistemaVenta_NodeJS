CREATE DATABASE IF NOT EXISTS `crudnodejs` 

CREATE TABLE empleados(  
    id int NOT NULL PRIMARY KEY AUTO_INCREMENT COMMENT 'Primary Key',
    DNI VARCHAR(8),
    NOMBRE VARCHAR(50),
    TELEFONO VARCHAR(9),
    DIRECCION VARCHAR(50),
    EMAIL VARCHAR(50),
    DISTRITO VARCHAR(50)

) DEFAULT CHARSET UTF8 COMMENT '';

CREATE TABLE clientes(  
    id int NOT NULL PRIMARY KEY AUTO_INCREMENT COMMENT 'Primary Key',
    NUMERO VARCHAR(11),
    NOMBRE VARCHAR(50),
    TELEFONO VARCHAR(9),
    DIRECCION VARCHAR(50),
    EMAIL VARCHAR(50),
    DISTRITO VARCHAR(50)

) DEFAULT CHARSET UTF8 COMMENT '';

CREATE TABLE proveedores(  
    id int NOT NULL PRIMARY KEY AUTO_INCREMENT COMMENT 'Primary Key',
    RUC VARCHAR(11),
    NOMBRE VARCHAR(50),
    TELEFONO VARCHAR(9),
    DIRECCION VARCHAR(50),
    EMAIL VARCHAR(50),
    DISTRITO VARCHAR(50)

) DEFAULT CHARSET UTF8 COMMENT '';

CREATE TABLE productos(  
    id int NOT NULL PRIMARY KEY AUTO_INCREMENT COMMENT 'Primary Key',
    DESCRIPCION VARCHAR(50),
    STOCK INT,
    PRECIO_VENTA DECIMAL(10,2),
    PRECIO_COMPRA DECIMAL(10,2),
    CLASIFICACION VARCHAR(50),
    MARCA VARCHAR(50)

) DEFAULT CHARSET UTF8 COMMENT '';
